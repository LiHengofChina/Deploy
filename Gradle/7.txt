Gradle学习系列之七――依赖管理

//----------------------------------------------添加maven依赖
build.gradle

repositories {
   mavenCentral()
}
//-------------------------------依赖分组
	//一组依赖
configurations {
   myDependency
}  
//------------------添加具体依赖的内容
dependencies {
   myDependency 'org.apache.commons:commons-lang3:3.0'
}  
//------------------task引入
task showMyDependency << {
   println configurations.myDependency.asPath
}
